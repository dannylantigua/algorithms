public Class FizzBuzz{
    public static void main (String args[]){
        for(int a = 1; a <= 100; a++){
            if ((a%3==1) && (a%5==1)){
                System.out.println("FizzBuzz");
            }else{
                if(a%3==1){
                    System.out.println("Fizz");
                }
                if(a%5==1){
                    System.out.println("Buzz");
                }
            }
        }
    }
}
